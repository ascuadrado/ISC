setup keyboard:
	echo XKBLAYOUT="es" | sudo tee -a /etc/default/keyboard
	sudo reboot

Install can-utils
	sudo apt-get install can-utils
Install python-can
	pip3 install python-can
Install sqlite3
	sudo apt-get install sqlite3
Install java (for processing)
	sudo apt install openjdk-8-jdk
Avoid screen sleep
	Add on: /etc/xdg/lxsession/LXDE-pi/autostart
		@xset s off
		@xset -dpms

Setup of HDMI Screen
	Add to /boot/config.txt:
		# HDMI Screen
		hdmi_group=2
		hdmi_mode=87
		hdmi_cvt=800 480 60 6 0 0 0
		hdmi_drive=1

max_usb_current=1

setup CAN:
	Add to /boot/config.txt:
		dtparam=spi=on
		dtoverlay=mcp2515-can0,oscillator=8000000,interrupt=25
		dtoverlay=mcp2515-can1,oscillator=8000000,interrupt=24
	or execute:
		echo dtparam=spi=on | sudo tee -a /boot/config.txt
		echo dtoverlay=mcp2515-can0,oscillator=8000000,interrupt=25 | sudo tee -a /boot/config.txt
		echo dtoverlay=mcp2515-can1,oscillator=8000000,interrupt=24 | sudo tee -a /boot/config.txt
Start CAN
	manual:
		sudo ip link set can0 up type can bitrate 250000
		sudo ip link set can1 up type can bitrate 500000
